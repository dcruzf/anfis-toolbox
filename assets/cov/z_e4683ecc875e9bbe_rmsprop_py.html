11def _zeros_like_structure(params): 

12 """Create a zero-structure matching model.get_parameters() format. 

13 

14 Returns a dict with: 

15 - 'consequent': np.zeros_like(params['consequent']) 

16 - 'membership': { name: [ {param_name: 0.0, ...} ] } 

17 """ 

18 out = {"consequent": np.zeros_like(params["consequent"]), "membership": {}} 

19 for name, mf_list in params["membership"].items(): 

20 out["membership"][name] = [] 

21 for mf_params in mf_list: 

22 out["membership"][name].append(dict.fromkeys(mf_params.keys(), 0.0)) 

23 return out

27class RMSPropTrainer(BaseTrainer): 

28 """RMSProp optimizer-based trainer for ANFIS. 

29 

30 Parameters: 

31 learning_rate: Base step size (alpha). 

32 rho: Exponential decay rate for the squared gradient moving average. 

33 epsilon: Small constant for numerical stability. 

34 epochs: Number of passes over the dataset. 

35 batch_size: If None, use full-batch; otherwise mini-batches of this size. 

36 shuffle: Whether to shuffle the data at each epoch when using mini-batches. 

37 verbose: Unused here; kept for API parity. 

38 

39 Notes: 

40 Supports configurable losses via the ``loss`` parameter. Defaults to mean squared error for 

41 regression tasks but can be switched to other differentiable objectives such as categorical 

42 cross-entropy when training ``ANFISClassifier`` models. 

43 """ 

44 

45 learning_rate: float = 0.001 

46 rho: float = 0.9 

47 epsilon: float = 1e-8 

48 epochs: int = 100 

49 batch_size: None | int = None 

50 shuffle: bool = True 

51 verbose: bool = False 

52 loss: LossFunction | str | None = None 

53 _loss_fn: LossFunction = field(init=False, repr=False)

55 def _prepare_training_data(self, model, X: np.ndarray, y: np.ndarray) -> tuple[np.ndarray, np.ndarray]: 

56 self._loss_fn = resolve_loss(self.loss) 

57 X_arr = np.asarray(X, dtype=float) 

58 y_arr = self._loss_fn.prepare_targets(y, model=model) 

59 if y_arr.shape[0] != X_arr.shape[0]: 

60 raise ValueError("Target array must have same number of rows as X")

80 def train_step(self, model, Xb: np.ndarray, yb: np.ndarray, state): 

81 """One RMSProp step on a batch; returns (loss, updated_state).""" 

82 loss, grads = self._compute_loss_and_grads(model, Xb, yb) 

83 self._apply_rmsprop_step(model, state["params"], state["cache"], grads) 

84 return loss, state

86 def _compute_loss_and_grads(self, model, Xb: np.ndarray, yb: np.ndarray) -> tuple[float, dict]: 

87 """Forward pass, MSE loss, backward pass, and gradients for a batch. 

88 

89 Returns (loss, grads) where grads follows model.get_gradients() structure. 

90 """ 

91 model.reset_gradients() 

92 y_pred = model.forward(Xb) 

93 loss = self._loss_fn.loss(yb, y_pred) 

94 dL_dy = self._loss_fn.gradient(yb, y_pred) 

95 model.backward(dL_dy) 

96 grads = model.get_gradients() 

97 return loss, grads

99 def _apply_rmsprop_step( 

100 self, 

101 model, 

102 params: dict, 

103 cache: dict, 

104 grads: dict, 

105 ) -> None: 

106 """Apply one RMSProp update to params using grads and caches. 

107 

108 Updates both consequent array parameters and membership scalar parameters. 

109 """ 

110 # Consequent is a numpy array 

111 g = grads["consequent"] 

112 c = cache["consequent"] 

113 c[:] = self.rho * c + (1.0 - self.rho) * (g * g) 

114 params["consequent"] = params["consequent"] - self.learning_rate * g / (np.sqrt(c) + self.epsilon) 

115 

116 # Membership are scalars in nested dicts 

117 for name in params["membership"].keys(): 

118 for i in range(len(params["membership"][name])): 

119 for key in params["membership"][name][i].keys(): 

120 gk = float(grads["membership"][name][i][key]) 

121 ck = cache["membership"][name][i][key] 

122 ck = self.rho * ck + (1.0 - self.rho) * (gk * gk) 

123 step = self.learning_rate * gk / (np.sqrt(ck) + self.epsilon) 

124 params["membership"][name][i][key] -= float(step) 

125 cache["membership"][name][i][key] = ck 

126 

127 # Push updated params back into the model 

128 model.set_parameters(params)

130 def compute_loss(self, model, X: np.ndarray, y: np.ndarray) -> float: 

131 """Return the current loss value for ``(X, y)`` without modifying state.""" 

132 preds = model.forward(X) 

133 return float(self._loss_fn.loss(y, preds))
